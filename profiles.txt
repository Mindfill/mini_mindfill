import asyncio
import logging
from datetime import date, datetime, timezone
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel
from supabase import AsyncClient

from app.auth.verify import get_current_user_id
from app.core.deps import get_supabase
from app.core.sanitize import sanitize_text

router = APIRouter()
logger = logging.getLogger(__name__)


class ProfileUpsert(BaseModel):
    full_name: str | None = None
    date_of_birth: date | None = None


@router.post("/profile")
async def upsert_profile(
    input: ProfileUpsert,
    user_id: str = Depends(get_current_user_id),
    supabase: AsyncClient = Depends(get_supabase)
):
    data = {}
    if input.full_name is not None:
        data["full_name"] = sanitize_text(input.full_name)
    if input.date_of_birth is not None:
        data["date_of_birth"] = input.date_of_birth.isoformat()

    if not data:
        raise HTTPException(status_code=400, detail="No fields provided to update")

    data["user_id"] = user_id
    data["updated_at"] = datetime.now(timezone.utc).isoformat()

    try:
        res = await (
            supabase.table("user_profiles")
            .upsert(data, on_conflict="user_id")
            .execute()
        )
    except Exception as e:
        logger.exception("Failed to upsert profile", extra={"user_id": user_id})
        raise HTTPException(status_code=500, detail=f"Failed to save profile: {e}")

    return res.data[0]


@router.get("/profile")
async def get_profile(
    user_id: str = Depends(get_current_user_id),
    supabase: AsyncClient = Depends(get_supabase)
):
    try:
        profile_res, auth_res = await asyncio.gather(
            supabase.table("user_profiles")
            .select("full_name, date_of_birth, created_at, updated_at")
            .eq("user_id", user_id)
            .maybe_single()
            .execute(),

            supabase.auth.admin.get_user_by_id(user_id)
        )
    except Exception as e:
        logger.exception("Failed to fetch profile", extra={"user_id": user_id})
        raise HTTPException(status_code=500, detail=f"Failed to fetch profile: {e}")

    email = auth_res.user.email if auth_res and auth_res.user else None
    profile = profile_res.data or {}

    return {
        "email": email,
        "full_name": profile.get("full_name"),
        "date_of_birth": profile.get("date_of_birth"),
    }
